double EMA(const double &price[], int period, int index)
{
   double alpha = 2.0 / (period + 1);
   double ema = price[index];
   for (int i = index + 1; i < ArraySize(price); i++)
      ema = alpha * price[i] + (1 - alpha) * ema;
   return ema;
}

double ComputeTEMA(const double &price[], int period, int index)
{
   double ema1 = EMA(price, period, index);
   double ema2 = EMA(price, period, index + 1); // Verschachtelt
   double ema3 = EMA(price, period, index + 2); // Noch tiefer
   return 3.0 * ema1 - 3.0 * ema2 + ema3;
}

🗑️ So löschst du eine Datei direkt auf GitHub:

  1. Gehe in deinem Repository auf die Datei, also z. B. https://github.com/DeinBenutzername/DeinRepoName/blob/main/Vorlagr/ZEMA_Evolution-ZEMA_Recorder.mq5
